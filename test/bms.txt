#include <Arduino.h>
#include "config/pin_map.hpp"
#include "utils/logger.hpp"
#include "drivers/bms_uart.hpp"

// raw echo test

static BmsUart bms({
  .serial = &Serial2,
  .baud = BMS_BAUD
});

void setup() {
  // Safety: disable motor
  pinMode(PIN_MOTOR_SLP, OUTPUT);
  digitalWrite(PIN_MOTOR_SLP, LOW);
  analogWrite(PIN_MOTOR_PWM, 0);

  logger::begin(115200);
  LOGI(F("BMS UART smoke test"));

  bms.begin();
}

void loop() {
  bms.update();

  if (bms.frameAvailable()) {
    uint8_t buf[256];
    uint16_t len = bms.readFrame(buf, sizeof(buf));

    Serial.print("RX len=");
    Serial.print(len);
    Serial.print(" data=");

    for (uint16_t i = 0; i < len; i++) {
      Serial.print(buf[i], HEX);
      Serial.print(" ");
    }
    Serial.println();
  }
}
