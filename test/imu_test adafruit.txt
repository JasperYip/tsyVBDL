#include <Arduino.h>
#include <Wire.h>
#include "Adafruit_BNO08x.h"

// BNO085 on ToF I2C port: SDA=17, SCL=16 (Wire1)
// platformio.ini lib_deps: adafruit/Adafruit BNO08x

static Adafruit_BNO08x imu;
static sh2_SensorValue_t sensorValue;
static uint32_t lastImuEvent = 0;

void initIMU()
{
    Wire1.end();
    delay(50);
    Wire1.setSDA(17);
    Wire1.setSCL(16);
    Wire1.setClock(400000);
    delay(100);

    while (!imu.begin_I2C(BNO08x_I2CADDR_DEFAULT, &Wire1)) {
        Serial.println("BNO085 not found - retrying...");
        Wire1.end();
        delay(200);
        Wire1.setSDA(17);
        Wire1.setSCL(16);
        Wire1.setClock(100000);
        delay(100);
    }
    Serial.println("BNO085 init OK");

    if (!imu.enableReport(SH2_ROTATION_VECTOR, 10000))
        Serial.println("WARN: Could not enable rotation vector");

    delay(100);
    lastImuEvent = millis();
}

void setup()
{
    Serial.begin(115200);
    while (!Serial) { delay(10); }
    Serial.println("BNO085 IMU Test");

    initIMU();
}

void loop()
{
    static uint32_t lastPrint = 0;
    static float roll = 0, pitch = 0, yaw = 0;

    if (imu.getSensorEvent(&sensorValue)) {
        lastImuEvent = millis();
        if (sensorValue.sensorId == SH2_ROTATION_VECTOR) {
            float w = sensorValue.un.rotationVector.real;
            float x = sensorValue.un.rotationVector.i;
            float y = sensorValue.un.rotationVector.j;
            float z = sensorValue.un.rotationVector.k;

            roll  = atan2f(2*(w*x + y*z), 1 - 2*(x*x + y*y)) * 180.0f / M_PI;
            pitch = asinf(constrain(2*(w*y - z*x), -1.0f, 1.0f)) * 180.0f / M_PI;
            yaw   = atan2f(2*(w*z + x*y), 1 - 2*(y*y + z*z)) * 180.0f / M_PI;
        }
    } else if (millis() - lastImuEvent > 2000) {
        Serial.println("BNO085 connection lost - restarting...");
        initIMU();
        return;
    } else {
        return;
    }

    if (millis() - lastPrint < 200) return;  // 5Hz
    lastPrint = millis();

    Serial.print("roll=");    Serial.print(roll,  1);
    Serial.print("  pitch="); Serial.print(pitch, 1);
    Serial.print("  yaw=");   Serial.println(yaw, 1);
}
