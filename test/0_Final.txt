bool isFrameDifferentIgnoreSeq(const CanFrame& a, const CanFrame& b, int seqIndex)
{
  if (a.id != b.id) return true;
  if (a.len != b.len) return true;

  for (uint8_t i = 0; i < a.len; i++) {
    if (i == seqIndex) continue;
    if (a.data[i] != b.data[i]) return true;
  }
  return false;
}

float percentToMm(float percent)
{
  percent = constrain(percent, 0.0f, 100.0f);
  return config::STROKE_HARD_MIN_MM +
         (percent / 100.0f) * config::STROKE_SPAN_MM;
}

float computeSoftZoneScale(float pos_mm)
{
  if (pos_mm <= config::STROKE_SOFT_START_MM) {
    float t = (pos_mm - config::STROKE_HARD_MIN_MM) /
              (config::STROKE_SOFT_START_MM - config::STROKE_HARD_MIN_MM);
    t = constrain(t, 0.0f, 1.0f);
    return config::SOFT_ZONE_MAX_DUTY +
           t * (1.0f - config::SOFT_ZONE_MAX_DUTY);
  }

  if (pos_mm >= config::STROKE_SOFT_END_MM) {
    float t = (config::STROKE_HARD_MAX_MM - pos_mm) /
              (config::STROKE_HARD_MAX_MM - config::STROKE_SOFT_END_MM);
    t = constrain(t, 0.0f, 1.0f);
    return config::SOFT_ZONE_MAX_DUTY +
           t * (1.0f - config::SOFT_ZONE_MAX_DUTY);
  }

  return 1.0f;
}

bool isHardLimitHit(float pos_mm, bool extending)
{
  if (extending) {
    return pos_mm >= config::STROKE_HARD_MAX_MM;
  }
  return pos_mm <= config::STROKE_HARD_MIN_MM;
}

const char* modeName(ControlMode m)
{
  switch (m) {
    case ControlMode::OVERRIDE:            return "OVERRIDE";
    case ControlMode::CAN:                return "CAN";
    case ControlMode::HOMING:             return "HOMING";
    case ControlMode::EMERGENCY_RESURFACE: return "EMERG_RESURFACE";
  }
  return "?";
}

uint8_t mapSystemState()
{
  if (lastSafety.hard_fault)                    return can::STATE_FAULT;
  if (mode == ControlMode::EMERGENCY_RESURFACE)  return can::STATE_FAULT;
  if (mode == ControlMode::HOMING)               return can::STATE_HOMING;
  if (!homed)                                    return can::STATE_UNHOMED;
  if (fabs(lastEst.vel_est_mm_s) > 0.5f)        return can::STATE_RUN;
  return can::STATE_HOLD;
}

void printStatus(const Estimator::Output& est)
{
  const auto& bms = bmsManager.telemetry();

  // ── Header: mode / control state ────────────────────────────────────────
  Serial.print("mode="); Serial.print(modeName(mode));
  Serial.print("  state=");
  switch (mapSystemState()) {
    case can::STATE_UNHOMED: Serial.print("UNHOMED"); break;
    case can::STATE_HOMING:  Serial.print("HOMING");  break;
    case can::STATE_RUN:     Serial.print("RUN");     break;
    case can::STATE_HOLD:    Serial.print("HOLD");    break;
    case can::STATE_FAULT:   Serial.print("FAULT");   break;
    default:                 Serial.print("?");       break;
  }
  Serial.print("  target="); Serial.print(target_mm, 1);
  Serial.print("mm("); Serial.print(target_percent, 0); Serial.print("%)");
  Serial.print("  err="); Serial.print(target_mm - est.pos_est_mm, 2); Serial.print("mm");
  Serial.println();

  // ── 0x050  STATUS_FAULT ─────────────────────────────────────────────────
  Serial.print("  0x050  HF_A=0x"); Serial.print(lastSafety.hard_fault_a, HEX);
  Serial.print(" HF_B=0x");         Serial.print(lastSafety.hard_fault_b, HEX);
  Serial.print(" SF=0x");           Serial.print(lastSafety.soft_fault_bits, HEX);
  Serial.print(" first=");          Serial.print(lastSafety.first_hard_fault);
  Serial.println();

  // ── 0x200  STATUS_CONTROL ───────────────────────────────────────────────
  uint8_t flags = 0;
  if (homed)                               flags |= can::FLAG_HOMED;
  if (fabs(est.vel_est_mm_s) > 0.5f)      flags |= can::FLAG_MOVING;
  if (motor.enabled())                     flags |= can::FLAG_MOTOR_EN;
  if (lastDirectionExtend)                 flags |= can::FLAG_DIR;
  if (motor.faultActive())                 flags |= can::FLAG_DRIVER_FLT;
  if (leak.isLeak())                       flags |= can::FLAG_LEAK;

  Serial.print("  0x200  pos="); Serial.print(est.pos_est_mm, 1); Serial.print("mm");
  Serial.print("  tof=");        Serial.print((int)lastTofMm); Serial.print("mm");
  Serial.print("  vel=");        Serial.print(est.vel_est_mm_s, 1); Serial.print("mm/s");
  Serial.print("  pwm=");        Serial.print(pwm);
  Serial.print("("); Serial.print((int)(pwm * 100 / PWM_MAX)); Serial.print("%)");
  Serial.print("  iA=");         Serial.print(motor.readCurrentA(), 2); Serial.print("A");
  Serial.print("  tofV=");       Serial.print(lastToF.valid ? "Y" : "N");
  Serial.print("  flags=0x");    Serial.print(flags, HEX);
  Serial.print("[");
  if (flags & can::FLAG_HOMED)      Serial.print("HMD ");
  if (flags & can::FLAG_MOVING)     Serial.print("MOV ");
  if (flags & can::FLAG_MOTOR_EN)   Serial.print("EN ");
  if (flags & can::FLAG_DIR)        Serial.print("EXT ");
  if (flags & can::FLAG_LEAK)       Serial.print("LEAK ");
  if (flags & can::FLAG_DRIVER_FLT) Serial.print("DRV_FLT ");
  Serial.print("]");
  Serial.println();

  // ── 0x210  STATUS_BMS ───────────────────────────────────────────────────
  Serial.print("  0x210  ");
  if (!USE_BMS) {
    Serial.print("BMS DISABLED");
  } else if (!bms.any_response) {
    Serial.print("NO SIGNAL");
  } else {
    Serial.print("pack="); Serial.print(bms.pack_voltage_v, 1); Serial.print("V");
    Serial.print("  curr="); Serial.print(bms.pack_current_a, 1); Serial.print("A");
    Serial.print("  min="); Serial.print(bms.min_cell_mv); Serial.print("mV");
    Serial.print("  max="); Serial.print(bms.max_cell_mv); Serial.print("mV");
    Serial.print("  T=");
    Serial.print(bms.pack_temp_dC / 10.0f, 0); Serial.print("C/");
    if (bms.ext1_temp_dC == -32768) Serial.print("n/c/"); else { Serial.print(bms.ext1_temp_dC / 10.0f, 0); Serial.print("C/"); }
    if (bms.ext2_temp_dC == -32768) Serial.print("n/c"); else { Serial.print(bms.ext2_temp_dC / 10.0f, 0); Serial.print("C"); }
    if (!bms.full_snapshot_ready) Serial.print("  (partial)");
  }
  Serial.println();

  // ── 0x220  STATUS_BMS_MORE ──────────────────────────────────────────────
  Serial.print("  0x220  ");
  if (!USE_BMS) {
    Serial.print("BMS DISABLED");
  } else if (!bms.any_response) {
    Serial.print("NO SIGNAL");
  } else {
    const char* stateStr = "?";
    switch (bms.online_status) {
      case 0x0091: stateStr = "Charging";     break;
      case 0x0092: stateStr = "FullCharged";  break;
      case 0x0093: stateStr = "Discharging";  break;
      case 0x0096: stateStr = "Regen";        break;
      case 0x0097: stateStr = "Idle";         break;
      case 0x009B: stateStr = "Fault";        break;
    }
    Serial.print("state="); Serial.print(stateStr);
    Serial.print("  cells=");
    if (bms.num_cells == 0) {
      Serial.print("waiting...");
    } else {
      for (uint8_t i = 0; i < bms.num_cells; i++) {
        Serial.print(bms.cell_mv[i]);
        if (i < bms.num_cells - 1) Serial.print("/");
      }
      Serial.print("mV");
    }
    const auto& bf = bmsManager.flags();
    if (bf.bms_temp_high || bf.bms_overcurrent || bf.bms_switch_fault ||
        bf.bms_low_voltage || bf.bms_high_voltage) {
      Serial.print("  BMS_FLT[");
      if (bf.bms_temp_high)    Serial.print("TEMP ");
      if (bf.bms_overcurrent)  Serial.print("OC ");
      if (bf.bms_switch_fault) Serial.print("SW ");
      if (bf.bms_low_voltage)  Serial.print("LOWV ");
      if (bf.bms_high_voltage) Serial.print("HIGHV ");
      Serial.print("]");
    }
  }
  Serial.println();
}

void handleCommand(String cmd)
{
  cmd.trim();
  cmd.toLowerCase();

  // === Always available regardless of mode ===
  if (cmd == "debug t") {
    debugPrint = true;
    Serial.println("Debug output ON");
    return;
  }
  if (cmd == "debug f") {
    debugPrint = false;
    Serial.println("Debug output OFF");
    return;
  }
  if (cmd == "help") {
    printHelp();
    return;
  }
  if (cmd == "status") {
    printStatus(lastEst);
    return;
  }

  // === Switch from CAN/HOMING to serial control ===
  if (cmd == "auto") {
    if (mode == ControlMode::HOMING) {
      Serial.println("Cannot switch mode during homing");
      return;
    }
    if (mode == ControlMode::EMERGENCY_RESURFACE) {
      Serial.println("WARNING: Cancelling emergency resurface - taking override control");
    }
    mode = ControlMode::OVERRIDE;
    override_use_pid = false;
    manual_pwm_cmd = 0;
    motor.setPWM(0);
    pwm = 0;
    Serial.println("OVERRIDE mode (PWM) - use '%<n>' for PID, 'help' for commands");
    return;
  }

  // === Block all motor/control commands while not in OVERRIDE ===
  if (mode != ControlMode::OVERRIDE) {
    Serial.println("In CAN/HOMING mode - send 'auto' to switch to override control");
    return;
  }

  // === OVERRIDE sub-mode selection ===
  if (cmd == "m") {
    override_use_pid = false;
    manual_pwm_cmd = 0;
    motor.setPWM(0);
    pwm = 0;
    Serial.println("OVERRIDE (PWM) - use p/f/b/s");
    return;
  }

  if (cmd == "a") {
    if (!homed) {
      Serial.println("ERROR: Not homed");
      return;
    }
    override_use_pid = true;
    target_mm = lastEst.pos_est_mm;
    target_mm_cmd = target_mm;
    pid.reset();
    holding = false;
    Serial.println("OVERRIDE (PID) - use '%<n>'");
    return;
  }

  if (cmd == "c") {
    if (!homed) {
      Serial.println("ERROR: Not homed");
      return;
    }
    mode = ControlMode::CAN;
    lastCanRx = millis();
    target_mm = lastEst.pos_est_mm;
    target_mm_cmd = target_mm;
    pid.reset();
    holding = false;
    Serial.println("CAN mode");
    return;
  }

  if (cmd.startsWith("%")) {
    if (!homed) {
      Serial.println("ERROR: Not homed");
      return;
    }
    float pct = cmd.substring(1).toFloat();
    target_percent = constrain(pct, 0.0f, 100.0f);
    target_mm_cmd = percentToMm(target_percent);
    override_use_pid = true;
    pid.reset();
    holding = false;
    Serial.print("PID target: ");
    Serial.print(target_percent);
    Serial.print("% -> ");
    Serial.println(target_mm_cmd);
    return;
  }

  if (cmd.startsWith("p")) {
    manual_pwm_cmd = constrain(cmd.substring(1).toInt(), 0, 255);
    override_use_pid = false;
    return;
  }

  if (cmd == "f") {
    motor.setDirection(MotorDriver::Direction::EXTEND);
    lastDirectionExtend = true;
    override_use_pid = false;
    return;
  }

  if (cmd == "b") {
    motor.setDirection(MotorDriver::Direction::RETRACT);
    lastDirectionExtend = false;
    override_use_pid = false;
    return;
  }

  if (cmd == "s") {
    manual_pwm_cmd = 0;
    pwm = 0;
    motor.setPWM(0);
    override_use_pid = false;
    return;
  }

  if (cmd == "z") {
    float home_mm = 10.0f;
    if (USE_ENCODER) {
      int32_t home_counts = roundf(home_mm / config::MM_PER_COUNT);
      enc.writeCounts(home_counts);
    }
    estimator.reset();
    estimator.setPositionMm(home_mm);
    if (USE_ENCODER) estimator.syncEncoder(enc.readCounts());
    pid.reset();
    homed = true;
    Serial.println("HOMED at 10mm");
    return;
  }

  if (cmd == "recover") {
    safety.reset();
    lastSafety = {};
    Serial.println("Safety manager reset requested");
    return;
  }

  if (cmd == "inj0") {
    clearInjectedFaults();
    Serial.println("All injections cleared");
    return;
  }

  if (cmd == "inj") {
    printInjections();
    return;
  }

  // Hard injections
  if (cmd == "leak1") { inj_leak = true; return; }
  if (cmd == "leak0") { inj_leak = false; return; }
  if (cmd == "drv1") { inj_driver_fault_raw = true; return; }
  if (cmd == "drv0") { inj_driver_fault_raw = false; return; }
  if (cmd == "hcmd1") { inj_cmd_timeout_large = true; return; }
  if (cmd == "hcmd0") { inj_cmd_timeout_large = false; return; }
  if (cmd == "hbto1") { inj_bms_timeout_large = true; return; }
  if (cmd == "hbto0") { inj_bms_timeout_large = false; return; }
  if (cmd == "htemp1") { inj_bms_temp_high = true; return; }
  if (cmd == "htemp0") { inj_bms_temp_high = false; return; }
  if (cmd == "hoc1") { inj_bms_overcurrent = true; return; }
  if (cmd == "hoc0") { inj_bms_overcurrent = false; return; }
  if (cmd == "hsw1") { inj_bms_switch_fault = true; return; } 
  if (cmd == "hsw0") { inj_bms_switch_fault = false; return; }
  if (cmd == "hhome1") { inj_homing_failed = true; return; }
  if (cmd == "hhome0") { inj_homing_failed = false; return; }

  // Soft injections
  if (cmd == "scmd1") { inj_cmd_timeout_small = true; return; }
  if (cmd == "scmd0") { inj_cmd_timeout_small = false; return; }
  if (cmd == "sbto1") { inj_bms_timeout_small = true; return; }
  if (cmd == "sbto0") { inj_bms_timeout_small = false; return; }
  if (cmd == "sblow1") { inj_bms_low_voltage = true; return; }
  if (cmd == "sblow0") { inj_bms_low_voltage = false; return; }
  if (cmd == "sbhigh1") { inj_bms_high_voltage = true; return; }
  if (cmd == "sbhigh0") { inj_bms_high_voltage = false; return; }
  if (cmd == "stof1") { inj_tof_invalid = true; return; }
  if (cmd == "stof0") { inj_tof_invalid = false; return; }
  if (cmd == "soor1") { inj_tof_out_of_range = true; return; }
  if (cmd == "soor0") { inj_tof_out_of_range = false; return; }
  if (cmd == "sslip1") { inj_slip = true; return; }
  if (cmd == "sslip0") { inj_slip = false; return; }

  if (cmd == "h") {
    motor.setPWM(0);
    motor.setDirection(MotorDriver::Direction::RETRACT);
    lastDirectionExtend = false;
    manual_pwm_cmd = 0;
    pwm = 0;
    homingTimer_s = 0.0f;
    homed = false;
    homingFailed = false;
    holding = false;
    homingReturnMode = ControlMode::OVERRIDE;
    mode = ControlMode::HOMING;
    Serial.println("HOMING: retracting until ToF reads 10mm...");
    return;
  }

  Serial.print("Unknown command: ");
  Serial.println(cmd);
}

void readSerialCommands()
{
  while (Serial.available()) {
    char c = Serial.read();
    if (c == '\n' || c == '\r') {
      if (cmdBuffer.length() > 0) {
        handleCommand(cmdBuffer);
        cmdBuffer = "";
      }
    } else {
      if (cmdBuffer.length() < CMD_BUFFER_MAX_LEN) {
        cmdBuffer += c;
      } else {
        cmdBuffer = "";
        Serial.println("ERROR: Command too long");
      }
    }
  }
}

void handleCan()
{
  CanFrame frame;
  while (canBus.read(frame)) {
    constexpr int SEQ_INDEX = 6;
    uint16_t id = frame.id & 0x7FF;

    if (debugPrint &&
        (!lastFrames[id].valid ||
         isFrameDifferentIgnoreSeq(frame, lastFrames[id].frame, SEQ_INDEX))) {

      Serial.print("RX id=0x");
      Serial.print(frame.id, HEX);
      Serial.print(" len=");
      Serial.print(frame.len);
      Serial.print(" data=");
      for (uint8_t i = 0; i < frame.len; i++) {
        Serial.print(frame.data[i]);
        Serial.print(" ");
      }
      Serial.println();

      lastFrames[id].frame = frame;
      lastFrames[id].valid = true;
    }

    if (frame.id == can::CMD_SETPOINT_BASE + CAN_NODE_ID) {
      lastCanRx = millis();

      can::CmdSetpoint msg;
      can::unpackCmdSetpoint(frame.data, msg);

      if (mode != ControlMode::CAN) continue;

      // Homing triggered by Pi
      if ((msg.command_mode & can::CMD_START_HOMING) && !homed) {
        motor.setPWM(0);
        motor.setDirection(MotorDriver::Direction::RETRACT);
        lastDirectionExtend = false;
        manual_pwm_cmd = 0;
        pwm = 0;
        homingTimer_s = 0.0f;
        homed = false;
        homingFailed = false;
        holding = false;
        homingReturnMode = ControlMode::CAN;
        mode = ControlMode::HOMING;
        Serial.println("CAN: homing started");
        continue;
      }

      if (!homed) continue;

      float pct = constrain(msg.stroke_percent, 0, 100);
      static uint8_t last_pct = 255;

      if (msg.stroke_percent != last_pct) {
        pid.reset();
        holding = false;
        last_pct = msg.stroke_percent;
      }

      target_percent = pct;
      target_mm_cmd = percentToMm(pct);
    }
  }
}

uint8_t buildControlFlags()
{
  uint8_t flags = 0;
  if (homed) flags |= can::FLAG_HOMED;
  if (fabs(lastEst.vel_est_mm_s) > 0.5f) flags |= can::FLAG_MOVING;
  if (motor.enabled()) flags |= can::FLAG_MOTOR_EN;
  if (lastDirectionExtend) flags |= can::FLAG_DIR;
  if (motor.faultActive() || inj_driver_fault_raw) flags |= can::FLAG_DRIVER_FLT;
  if (leak.isLeak() || inj_leak) flags |= can::FLAG_LEAK;
  return flags;
}

void sendStatusControl()
{
  can::StatusControl msg{};
  msg.est_position_01mm = (uint16_t)constrain(lastEst.pos_est_mm * 100.0f, 0, 65535);
  msg.tof_position_mm = (uint16_t)constrain(lastTofMm, 0, 65535);
  msg.flags = buildControlFlags();
  msg.pwm = (uint8_t)pwm;
  msg.sequence = canSequenceControl++;

  float currentA = motor.readCurrentA();
  msg.motor_current = (int8_t)constrain(currentA * 10.0f, -127, 127);

  uint8_t buf[8];
  can::packStatusControl(buf, msg);

  CanFrame frame;
  frame.id = can::STATUS_CONTROL_BASE + CAN_NODE_ID;
  frame.len = 8;
  for (int i = 0; i < 8; i++) frame.data[i] = buf[i];
  canBus.write(frame);
}

// Helper: encode mV to the (mV−2500)/10 uint8 used in STATUS_BMS / STATUS_BMS_MORE.
// Clamps to valid uint8 range.
static uint8_t encodeCellV(uint16_t mv)
{
    if (mv < 2500) return 0;
    uint32_t raw = (mv - 2500u) / 10u;
    return (raw > 254u) ? 254u : static_cast<uint8_t>(raw);
}

// Map TinyBMS online_status word to compact 0-5 enum.
static uint8_t mapBmsStatus(uint16_t s)
{
    switch (s) {
        case 0x0091: return 1;  // Charging
        case 0x0092: return 2;  // FullyCharged
        case 0x0093: return 3;  // Discharging
        case 0x0096: return 3;  // Regeneration → treat as Discharging
        case 0x0097: return 4;  // Idle
        case 0x009B: return 5;  // Fault
        default:     return 0;  // Unknown
    }
}

void sendStatusBms()
{
    const auto& t = bmsManager.telemetry();

    can::StatusBMS msg{};

    // Pack voltage: float V → uint8 ×0.1V  (clamp 0–25.5V)
    {
        float v = t.pack_voltage_v * 10.0f;
        if (v < 0.0f) v = 0.0f;
        if (v > 255.0f) v = 255.0f;
        msg.pack_V = static_cast<uint8_t>(v);
    }

    msg.min_cell_V = encodeCellV(t.min_cell_mv);
    msg.max_cell_V = encodeCellV(t.max_cell_mv);

    // Pack current: float A → int8 ×0.5A  (+ = discharge)
    {
        float c = t.pack_current_a / 0.5f;
        if (c < -128.0f) c = -128.0f;
        if (c >  127.0f) c =  127.0f;
        msg.pack_current = static_cast<int8_t>(c);
    }

    // Temperature: 0.1°C int16 → uint8 °C  (0xFF = invalid / not connected)
    auto encodeTemp = [](int16_t dC) -> uint8_t {
        if (dC == -32768) return 0xFF;        // not connected sentinel
        int32_t degC = dC / 10;
        if (degC < 0)   return 0;
        if (degC > 254) return 254;
        return static_cast<uint8_t>(degC);
    };

    msg.temp_internal = encodeTemp(t.pack_temp_dC);
    msg.temp_ext1     = encodeTemp(t.ext1_temp_dC);
    msg.temp_ext2     = encodeTemp(t.ext2_temp_dC);

    msg.sequence = canSequenceBms++;

    uint8_t buf[8];
    can::packStatusBMS(buf, msg);

    CanFrame frame;
    frame.id  = can::STATUS_BMS_BASE + CAN_NODE_ID;
    frame.len = 8;
    for (int i = 0; i < 8; i++) frame.data[i] = buf[i];
    canBus.write(frame);
}

void sendStatusBmsMore()
{
    const auto& t = bmsManager.telemetry();

    can::StatusBmsMore msg{};

    const uint8_t cells = (t.num_cells > 6) ? 6 : t.num_cells;
    for (uint8_t i = 0; i < cells; i++) {
        msg.cell_V[i] = encodeCellV(t.cell_mv[i]);
    }
    // Unused cell slots stay 0.

    msg.online_status = mapBmsStatus(t.online_status);
    msg.sequence      = canSequenceBmsMore++;

    uint8_t buf[8];
    can::packStatusBmsMore(buf, msg);

    CanFrame frame;
    frame.id  = can::STATUS_BMS_MORE_BASE + CAN_NODE_ID;
    frame.len = 8;
    for (int i = 0; i < 8; i++) frame.data[i] = buf[i];
    canBus.write(frame);
}

// -----------------------------
// Homing (ToF-based)
// -----------------------------
void handleHoming(float dt)
{
  homingTimer_s += dt;

  if (homingTimer_s > config::HOMING_TIMEOUT_S) {
    motor.setPWM(0);
    pwm = 0;
    homingFailed = true;
    mode = homingReturnMode;
    Serial.println("HOMING TIMEOUT - aborted");
    return;
  }

  // Wait for a plausible ToF reading before moving
  if (lastTofMm < config::TOF_MIN_VALID_MM) {
    motor.setPWM(0);
    pwm = 0;
    return;
  }

  // Home condition: ToF at or inside the hard minimum
  if (lastTofMm <= config::STROKE_HARD_MIN_MM) {
    motor.setPWM(0);
    pwm = 0;

    float home_mm = config::STROKE_HARD_MIN_MM;
    if (USE_ENCODER) {
      int32_t home_counts = (int32_t)roundf(home_mm / config::MM_PER_COUNT);
      enc.writeCounts(home_counts);
    }
    estimator.reset();
    estimator.setPositionMm(home_mm);
    if (USE_ENCODER) estimator.syncEncoder(enc.readCounts());
    pid.reset();

    homed = true;
    mode = homingReturnMode;

    Serial.print("HOMED - ToF=");
    Serial.print(lastTofMm, 1);
    Serial.println(USE_ENCODER ? "mm, encoder set to 10mm" : "mm (ToF-only mode)");
    return;
  }

  // Soft zone scales duty down as we approach 10mm (10-20mm zone)
  float scale = computeSoftZoneScale(lastTofMm);
  float duty  = config::HOMING_PWM * scale;
  pwm = constrain((int)(duty * PWM_MAX), 1, PWM_MAX);

  motor.setDirection(MotorDriver::Direction::RETRACT);
  lastDirectionExtend = false;
  motor.setPWM(pwm);
}

// -----------------------------
// Setup
// -----------------------------
void setup()
{
  Serial.begin(115200);
  logger::begin(115200);
  cmdBuffer.reserve(CMD_BUFFER_MAX_LEN);

  pinMode(LED_BUILTIN, OUTPUT);
  digitalWrite(LED_BUILTIN, LOW);

  if (USE_ENCODER) enc.begin();
  leak.begin(false); // Blue Robotics SOS style output: active high

  estimator.reset();
  if (USE_ENCODER) estimator.syncEncoder(enc.readCounts());

  if (!tof.begin(Wire1)) {
    Serial.println("WARN: ToF init failed");
  } else {
    Serial.println("ToF init OK");
  }

  motor.begin();
  motor.enable();

  pid.configure({
    config::PID_KP,
    config::PID_KI,
    config::PID_KD,
    config::PID_I_LIMIT,
    config::PID_OUT_LIMIT
  });

  safety.configure({
    .overcurrent_hard_limit_a = config::OVERCURRENT_HARD_LIMIT_A,
    .overcurrent_confirm_s    = config::OVERCURRENT_CONFIRM_S,
    .stall_current_a          = config::STALL_CURRENT_A,
    .stall_min_duty           = config::STALL_MIN_DUTY,
    .stall_pos_eps_mm         = config::STALL_POS_EPS_MM,
    .stall_confirm_s          = config::STALL_CONFIRM_S,
    .bms_temp_confirm_s       = config::BMS_TEMP_CONFIRM_S,
    .stroke_real_min_mm       = config::STROKE_REAL_MIN_MM,
    .stroke_real_max_mm       = config::STROKE_REAL_MAX_MM,
    .leak_debounce_s          = config::LEAK_DEBOUNCE_S,
    .driver_fault_debounce_s  = config::DRIVER_FAULT_DEBOUNCE_S
  });
  safety.reset();

  canBus.begin(500000, PIN_CAN_R, PIN_CAN_D);

  if (USE_BMS) {
    bmsManager.configure({
        .transport           = &bmsUart,
        .poll_interval_ms    = 140,       // 8 commands × 140ms ≈ 1.1s full cycle
        .response_timeout_ms = 100,
        .max_retries         = 1,
        .fault_confirm_ms    = (uint32_t)(config::BMS_TEMP_CONFIRM_S * 1000.0f),
        .timeout_small_ms    = config::BMS_TIMEOUT_SMALL_MS,
        .timeout_large_ms    = config::BMS_TIMEOUT_LARGE_MS
    });
    bmsManager.begin(millis());
  }

  Serial.println("VBD ready - CAN mode. Send CMD_START_HOMING via CAN to home.");
  Serial.println("Serial: 'auto' = serial control, 'debug t/f' = status on/off, 'help' = commands");
}

// -----------------------------
// Loop
// -----------------------------
void loop()
{
  uint32_t now = millis();

  // LED heartbeat
  if (!ledOn && (now - lastLedToggle >= 5000)) {
    digitalWrite(LED_BUILTIN, HIGH);
    ledOn = true;
    lastLedToggle = now;
  }
  if (ledOn && (now - lastLedToggle >= 100)) {
    digitalWrite(LED_BUILTIN, LOW);
    ledOn = false;
  }

  // dt
  static uint32_t lastTime = 0;
  float dt = (lastTime == 0) ? 0.0f : (now - lastTime) / 1000.0f;
  lastTime = now;

  // If the loop hiccups badly, drop commanded output immediately.
  if (dt > 0.2f) {
    manual_pwm_cmd = 0;
    pwm = 0;
    holding = false;
    pid.reset();
    motor.setPWM(0);
  }

  // Sensors
  int32_t counts = USE_ENCODER ? enc.readCounts() : 0;

  lastToF = tof.read();
  lastTofRawMm = lastToF.mm;
  lastTofMm = lastToF.mm;

  // Estimator fuses only on fresh + valid samples.
  bool tof_valid_estimator = lastToF.valid && lastToF.fresh;

  lastEst = estimator.update(
    counts,
    tof_valid_estimator,
    lastToF.mm,
    dt
  );

  // Inputs
  readSerialCommands();
  handleCan();
  if (USE_BMS) bmsManager.update(now);

  // Emergency resurface: large CAN timeout only — not triggered if a real hard fault exists
  // (real faults stop motor and Pi coordinates; this is only the Pi-is-dead fallback)
  if (mode == ControlMode::CAN && lastCanRx != 0 &&
      (millis() - lastCanRx > CAN_TIMEOUT_LARGE_MS) &&
      !lastSafety.hard_fault) {
    mode = ControlMode::EMERGENCY_RESURFACE;
    target_mm_cmd = config::STROKE_HARD_MAX_MM;
    target_mm = lastEst.pos_est_mm;
    pid.reset();
    holding = false;
    Serial.println("!! EMERGENCY RESURFACE - CAN lost >10s, extending to 100%");
  }

  // -----------------------------
  // Motion logic
  // -----------------------------
  if (mode == ControlMode::HOMING) {
    handleHoming(dt);

  } else if (mode == ControlMode::OVERRIDE) {
    float pos = lastEst.pos_est_mm;

    if (!override_use_pid) {
      // --- Raw PWM control ---
      bool extending = lastDirectionExtend;
      float duty = static_cast<float>(manual_pwm_cmd) / static_cast<float>(PWM_MAX);
      duty *= computeSoftZoneScale(pos);
      if (isHardLimitHit(pos, extending)) duty = 0.0f;
      duty = constrain(duty, 0.0f, 1.0f);
      pwm = static_cast<int>(roundf(duty * static_cast<float>(PWM_MAX)));
      motor.setDirection(extending ?
        MotorDriver::Direction::EXTEND :
        MotorDriver::Direction::RETRACT);
      motor.setPWM(pwm);

    } else {
      // --- PID position control ---
      float vel = lastEst.vel_est_mm_s;
      float max_step = 40.0f * dt;
      float delta = target_mm_cmd - target_mm;
      if (fabs(delta) > max_step) {
        target_mm += (delta > 0 ? max_step : -max_step);
      } else {
        target_mm = target_mm_cmd;
      }
      target_mm = constrain(target_mm, config::STROKE_HARD_MIN_MM, config::STROKE_HARD_MAX_MM);

      float error = target_mm - pos;
      bool skipPid = false;

      if (holding) {
        if (fabs(error) > config::DRIFT_RESTART_MM) {
          holding = false;
        } else {
          pwm = 0; motor.setPWM(0); skipPid = true;
        }
      }

      if (!skipPid) {
        bool near_target = fabs(error) <= config::POS_TOL_MM;
        bool slow = fabs(vel) < 0.5f;
        bool not_pushing = (error > 0 && vel <= 0) || (error < 0 && vel >= 0);
        if (near_target && slow && not_pushing) {
          pwm = 0; motor.setPWM(0); holding = true; skipPid = true;
        }
      }

      if (!skipPid) {
        float effort = pid.update(target_mm, pos, vel, dt);
        bool dir = (effort > 0);
        lastDirectionExtend = dir;
        motor.setDirection(dir ? MotorDriver::Direction::EXTEND : MotorDriver::Direction::RETRACT);
        float duty = fabs(effort);
        if (duty > 0.0f) duty = config::MIN_DUTY + (1.0f - config::MIN_DUTY) * duty;
        duty *= computeSoftZoneScale(pos);
        if (duty > 0.0f && duty < config::MIN_DUTY) duty = config::MIN_DUTY;
        if (isHardLimitHit(pos, dir)) duty = 0.0f;
        // Raw ToF backstop: only fires when filtered pos is already near the limit —
        // a mid-range noise spike can't trigger this and cause a spurious cutoff.
        if (lastToF.valid && dir  && pos >= config::STROKE_HARD_MAX_MM - config::TOF_BACKSTOP_APPROACH_MM && lastTofMm >= config::STROKE_HARD_MAX_MM) duty = 0.0f;
        if (lastToF.valid && !dir && pos <= config::STROKE_HARD_MIN_MM + config::TOF_BACKSTOP_APPROACH_MM && lastTofMm <= config::STROKE_HARD_MIN_MM) duty = 0.0f;
        duty = constrain(duty, 0.0f, 1.0f);
        pwm = duty * PWM_MAX;
        motor.setPWM(pwm);
      }
    }

  } else if (mode == ControlMode::CAN || mode == ControlMode::EMERGENCY_RESURFACE) {
    if (mode == ControlMode::CAN && !homed) {
      motor.setPWM(0);
      pwm = 0;
    } else {

    float pos = lastEst.pos_est_mm;
    float vel = lastEst.vel_est_mm_s;

    float max_step = 40.0f * dt;
    float delta = target_mm_cmd - target_mm;
    if (fabs(delta) > max_step) {
      target_mm += (delta > 0 ? max_step : -max_step);
    } else {
      target_mm = target_mm_cmd;
    }
    target_mm = constrain(target_mm, config::STROKE_HARD_MIN_MM, config::STROKE_HARD_MAX_MM);

    float error = target_mm - pos;
    bool skipPid = false;

    if (holding) {
      if (fabs(error) > config::DRIFT_RESTART_MM) {
        holding = false;
      } else {
        pwm = 0; motor.setPWM(0); skipPid = true;
      }
    }

    if (!skipPid) {
      bool near_target = fabs(error) <= config::POS_TOL_MM;
      bool slow = fabs(vel) < 0.5f;
      bool not_pushing = (error > 0 && vel <= 0) || (error < 0 && vel >= 0);
      if (near_target && slow && not_pushing) {
        pwm = 0; motor.setPWM(0); holding = true; skipPid = true;
      }
    }

    if (!skipPid) {
      float effort = pid.update(target_mm, pos, vel, dt);
      bool dir = (effort > 0);
      lastDirectionExtend = dir;
      motor.setDirection(dir ? MotorDriver::Direction::EXTEND : MotorDriver::Direction::RETRACT);
      float duty = fabs(effort);
      if (duty > 0.0f) duty = config::MIN_DUTY + (1.0f - config::MIN_DUTY) * duty;
      duty *= computeSoftZoneScale(pos);
      if (duty > 0.0f && duty < config::MIN_DUTY) duty = config::MIN_DUTY;
      if (isHardLimitHit(pos, dir)) duty = 0.0f;
      // Raw ToF backstop: only fires when filtered pos is already near the limit —
      // a mid-range noise spike can't trigger this and cause a spurious cutoff.
      if (lastToF.valid && dir  && pos >= config::STROKE_HARD_MAX_MM - config::TOF_BACKSTOP_APPROACH_MM && lastTofMm >= config::STROKE_HARD_MAX_MM) duty = 0.0f;
      if (lastToF.valid && !dir && pos <= config::STROKE_HARD_MIN_MM + config::TOF_BACKSTOP_APPROACH_MM && lastTofMm <= config::STROKE_HARD_MIN_MM) duty = 0.0f;
      duty = constrain(duty, 0.0f, 1.0f);
      pwm = duty * PWM_MAX;
      motor.setPWM(pwm);
    }

    } // end homed/EMERGENCY_RESURFACE branch
  }

  // -----------------------------
  // Build Safety inputs + update
  // -----------------------------
  SafetyManager::Inputs sin{};
  sin.leak_raw = leak.isLeak() || inj_leak;
  sin.driver_fault_raw = motor.faultActive() || inj_driver_fault_raw;

  sin.motor_current_a = motor.readCurrentA();
  sin.motor_enabled = motor.enabled();
  sin.motor_duty = constrain((float)pwm / (float)PWM_MAX, 0.0f, 1.0f);

  sin.pos_est_mm = lastEst.pos_est_mm;
  sin.pos_valid  = homed;
  sin.homing_failed = homingFailed || inj_homing_failed;

  bool in_can_family = (mode == ControlMode::CAN ||
                        mode == ControlMode::EMERGENCY_RESURFACE);
  bool can_small_to = (in_can_family && lastCanRx != 0 &&
                       (millis() - lastCanRx > CAN_TIMEOUT_SMALL_MS));
  sin.cmd_timeout_small = can_small_to || inj_cmd_timeout_small;
  // Large timeout still latches as hard fault so Pi sees FAULT_CMD_TO_L on reconnect.
  // Motor disable is suppressed separately during emergency resurface.
  bool can_large_to = (mode == ControlMode::EMERGENCY_RESURFACE) ||
                      (mode == ControlMode::CAN && lastCanRx != 0 &&
                       (millis() - lastCanRx > CAN_TIMEOUT_LARGE_MS));
  sin.cmd_timeout_large = can_large_to || inj_cmd_timeout_large;

  if (USE_BMS) {
    const auto& bmsFlags = bmsManager.flags();
    sin.bms_timeout_small = bmsFlags.bms_timeout_small || inj_bms_timeout_small;
    sin.bms_timeout_large = bmsFlags.bms_timeout_large || inj_bms_timeout_large;
    sin.bms_temp_high     = bmsFlags.bms_temp_high     || inj_bms_temp_high;
    sin.bms_overcurrent   = bmsFlags.bms_overcurrent   || inj_bms_overcurrent;
    sin.bms_switch_fault  = bmsFlags.bms_switch_fault  || inj_bms_switch_fault;
    sin.bms_low_voltage   = bmsFlags.bms_low_voltage   || inj_bms_low_voltage;
    sin.bms_high_voltage  = bmsFlags.bms_high_voltage  || inj_bms_high_voltage;
  }
  // When USE_BMS = false all BMS safety inputs stay false (zero-initialised above).

  sin.tof_valid = inj_tof_invalid ? false : lastToF.valid;
  bool tof_oor = lastToF.valid &&
                 (lastTofMm < config::STROKE_REAL_MIN_MM ||
                  lastTofMm > config::STROKE_REAL_MAX_MM);
  sin.tof_out_of_range = tof_oor || inj_tof_out_of_range;
  sin.slip_detected = inj_slip || lastEst.slip_detected;

  lastSafety = safety.update(sin, dt);

  // Hard fault response: stop motor immediately.
  // Exception: during emergency resurface, suppress disable if the ONLY latched
  // fault is FAULT_CMD_TO_L — that fault is what caused the resurface, and the
  // motor must keep running to surface. Any other concurrent hard fault still wins.
  if (lastSafety.hard_fault) {
    bool only_can_timeout = (lastSafety.hard_fault_a == can::FAULT_CMD_TO_L &&
                             lastSafety.hard_fault_b == 0);
    bool suppress_disable = (mode == ControlMode::EMERGENCY_RESURFACE && only_can_timeout);

    if (!suppress_disable) {
      motor.setPWM(0);
      motor.disable();
      manual_pwm_cmd = 0;
      pwm = 0;
    } else {
      if (!motor.enabled()) motor.enable();
    }
  } else {
    if (!motor.enabled()) {
      motor.enable();
    }
  }

  // -----------------------------
  // CAN TX
  // -----------------------------
  if (now - lastCanTxControl >= CAN_TX_CONTROL_INTERVAL_MS) {
    lastCanTxControl = now;
    sendStatusControl();
  }

  if (now - lastCanTxFault >= CAN_TX_FAULT_INTERVAL_MS) {
    lastCanTxFault = now;
    sendStatusFault();
  }

  if (USE_BMS && (now - lastCanTxBms >= CAN_TX_BMS_INTERVAL_MS)) {
    lastCanTxBms = now;
    sendStatusBms();
    sendStatusBmsMore();
  }

  // -----------------------------
  // Status print (only when debug enabled)
  // -----------------------------
  if (debugPrint && (now - lastPrint >= 2000)) {
    lastPrint = now;
    printStatus(lastEst);
  }
}
