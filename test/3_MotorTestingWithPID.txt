// --- Encoder ---
static EncoderAlt enc({
  .channel = 2,

  .pin_a = PIN_ENC_A,
  .pin_b = PIN_ENC_B,

  .counts_per_rev = config::COUNTS_PER_SCREW_REV,
  .lead_screw_pitch_mm = config::LEAD_SCREW_MM_PER_REV
});

// --- Motor ---
static MotorDriver motor({
  .pin_pwm = PIN_MOTOR_PWM,
  .pin_dir = PIN_MOTOR_DIR,
  .pin_slp = PIN_MOTOR_SLP,
  .pin_flt = PIN_MOTOR_FLT,
  .pin_cs  = PIN_MOTOR_CS,
  .pwm_freq_hz = 25000.0f,
  .pwm_max = 255,
  .flt_active_low = true,
  .cs_volts_per_amp = 0.04f,
  .adc_ref_volts = 3.3f,
  .adc_bits = 12
});

// --- Estimator ---
static Estimator estimator({
  .mm_per_count      = config::MM_PER_COUNT,
  .tof_gain          = config::TOF_FUSION_GAIN,
  .recovery_gain     = config::SLIP_RECOVERY_GAIN,
  .slip_detect_mm    = config::SLIP_DETECT_MM,
  .tof_min_valid_mm  = config::TOF_MIN_VALID_MM
});

// --- PID ---
static PIDController pid;

// --- ToF ---
static ToFSensor tof(PIN_TOF_SDA, PIN_TOF_SCL);

enum class ControlMode {
  AUTO,
  MANUAL
};

ControlMode mode = ControlMode::MANUAL;

int pwm = 0;               // applied PWM after safety shaping
int manual_pwm_cmd = 0;    // raw manual command from serial
const int PWM_MAX = 255;

bool forward = true;

uint32_t lastEstimatorUpdate = 0;
uint32_t lastControlUpdate = 0;
uint32_t lastPrint = 0;

const uint32_t CONTROL_INTERVAL_MS = 2; // 500Hz

Estimator::Output lastEst{};
ToFSensor::Reading lastToF{};

// --- Control state ---
float target_mm_cmd = config::STROKE_HARD_MIN_MM;
float target_mm = target_mm_cmd;       // filtered target
float target_percent = 0.0f;

bool homed = false;
bool holding = false;
bool posHardFaultLatched = false;

// Serial buffer
String cmdBuffer;
const uint16_t CMD_BUFFER_MAX_LEN = 64;

// -----------------------------
// Helpers
// -----------------------------
float percentToMm(float percent)
{
  percent = constrain(percent, 0.0f, 100.0f);

  return config::STROKE_HARD_MIN_MM +
         (percent / 100.0f) * config::STROKE_SPAN_MM;
}

float computeSoftZoneScale(float pos_mm)
{
  if (pos_mm <= config::STROKE_SOFT_START_MM) {
    float t = (pos_mm - config::STROKE_HARD_MIN_MM) /
              (config::STROKE_SOFT_START_MM - config::STROKE_HARD_MIN_MM);
    t = constrain(t, 0.0f, 1.0f);

    return config::SOFT_ZONE_MAX_DUTY +
           t * (1.0f - config::SOFT_ZONE_MAX_DUTY);
  }

  if (pos_mm >= config::STROKE_SOFT_END_MM) {
    float t = (config::STROKE_HARD_MAX_MM - pos_mm) /
              (config::STROKE_HARD_MAX_MM - config::STROKE_SOFT_END_MM);
    t = constrain(t, 0.0f, 1.0f);

    return config::SOFT_ZONE_MAX_DUTY +
           t * (1.0f - config::SOFT_ZONE_MAX_DUTY);
  }

  return 1.0f;
}

bool isHardLimitHit(float pos_mm, bool extending)
{
  if (extending) {
    return pos_mm >= config::STROKE_HARD_MAX_MM;
  }

  return pos_mm <= config::STROKE_HARD_MIN_MM;
}

void printHelp() {
  Serial.println("\n=== Commands ===");
  Serial.println("m          -> manual mode");
  Serial.println("a          -> auto mode");
  Serial.println("%<num>     -> set target (AUTO only)");
  Serial.println("p<num>     -> set PWM (manual)");
  Serial.println("f / b      -> direction (manual)");
  Serial.println("s          -> stop");
  Serial.println("x          -> clear latched position hard fault");
  Serial.println("z          -> home (must be near 10mm)");
  Serial.println("status     -> print status\n");
}

void printStatus(const Estimator::Output& est) {
  float error = target_mm - est.pos_est_mm;
  float motor_current_a = motor.readCurrentA();
  bool flt_active = motor.faultActive();

  Serial.print("mode=");
  Serial.print(mode == ControlMode::AUTO ? "AUTO" : "MANUAL");

  Serial.print("  mm=");
  Serial.print(est.pos_est_mm, 3);

  Serial.print("  target=");
  Serial.print(target_mm, 2);

  Serial.print("  target%=");
  Serial.print(target_percent, 1);

  Serial.print("  err=");
  Serial.print(error, 3);

  Serial.print("  vel=");
  Serial.print(est.vel_est_mm_s, 2);

  Serial.print("  pwm=");
  Serial.print(pwm);

  Serial.print("  flt=");
  Serial.print(flt_active ? "Y" : "N");

  Serial.print("  iA=");
  Serial.print(motor_current_a, 2);

  Serial.print("  tof=");
  Serial.print(lastToF.mm);

  Serial.print("  tofSt=");
  Serial.print(lastToF.status);

  Serial.print("  tofV=");
  Serial.print(lastToF.valid ? "Y" : "N");

  Serial.print("  tofF=");
  Serial.print(lastToF.fresh ? "Y" : "N");

  Serial.print("  hardLatch=");
  Serial.print(posHardFaultLatched ? "Y" : "N");

  Serial.print("  homed=");
  Serial.print(homed ? "Y" : "N");

  Serial.println();
}

// -----------------------------
// Command handling
// -----------------------------
void handleCommand(String cmd) {
  cmd.trim();
  cmd.toLowerCase();

  if (cmd == "m") {
    mode = ControlMode::MANUAL;
    manual_pwm_cmd = 0;
    pwm = 0;
    motor.setPWM(0);
    Serial.println("MANUAL mode");
  }
  else if (cmd == "a") {
    if (!homed) {
      Serial.println("ERROR: Not homed");
      return;
    }

    mode = ControlMode::AUTO;
    target_mm = lastEst.pos_est_mm;
    target_mm_cmd = target_mm;
    pid.reset();
    holding = false;

    Serial.println("AUTO mode");
  }
  else if (cmd.startsWith("%")) {
    if (mode != ControlMode::AUTO) {
      Serial.println("ERROR: AUTO mode only");
      return;
    }

    float pct = cmd.substring(1).toFloat();
    target_percent = constrain(pct, 0.0f, 100.0f);
    target_mm_cmd = percentToMm(target_percent);

    Serial.print("Target: ");
    Serial.print(target_percent);
    Serial.print("% -> ");
    Serial.println(target_mm_cmd);
  }
  else if (cmd.startsWith("p")) {
    if (mode == ControlMode::AUTO) return;

    manual_pwm_cmd = constrain(cmd.substring(1).toInt(), 0, 255);
  }
  else if (cmd == "f") {
    if (mode == ControlMode::AUTO) return;
    forward = true;
    motor.setDirection(MotorDriver::Direction::EXTEND);
  }
  else if (cmd == "b") {
    if (mode == ControlMode::AUTO) return;
    forward = false;
    motor.setDirection(MotorDriver::Direction::RETRACT);
  }
  else if (cmd == "s") {
    manual_pwm_cmd = 0;
    pwm = 0;
    motor.setPWM(0);
  }
  else if (cmd == "x") {
    if (motor.faultActive()) {
      Serial.println("ERROR: Driver fault still active");
      return;
    }

    if (lastEst.pos_est_mm < config::STROKE_REAL_MIN_MM ||
        lastEst.pos_est_mm > config::STROKE_REAL_MAX_MM) {
      Serial.println("ERROR: Position still outside 0..150mm");
      return;
    }

    posHardFaultLatched = false;
    motor.enable();
    Serial.println("Position hard fault latch cleared");
  }
  else if (cmd == "z") {
    float pos = lastEst.pos_est_mm;

    // if (fabs(pos - 10.0f) > 2.0f) {
    //   Serial.println("ERROR: Not near 10mm");
    //   return;
    // }

    float home_mm = 10.0f;
    int32_t home_counts = roundf(home_mm / config::MM_PER_COUNT);

    enc.writeCounts(home_counts);
    estimator.reset();
    estimator.setPositionMm(home_mm);
    estimator.syncEncoder(enc.readCounts());

    pid.reset();
    homed = true;

    Serial.println("HOMED at 10mm");
  }
}

void readSerialCommands() {
  while (Serial.available()) {
    char c = Serial.read();

    if (c == '\n' || c == '\r') {
      handleCommand(cmdBuffer);
      cmdBuffer = "";
    } else {
      if (cmdBuffer.length() < CMD_BUFFER_MAX_LEN) {
        cmdBuffer += c;
      } else {
        cmdBuffer = "";
        Serial.println("ERROR: Command too long");
      }
    }
  }
}

// -----------------------------
// Setup
// -----------------------------
void setup() {
  Serial.begin(115200);
  logger::begin(115200);
  cmdBuffer.reserve(CMD_BUFFER_MAX_LEN);

  enc.begin();

  estimator.reset();
  estimator.syncEncoder(enc.readCounts());  // IMPORTANT

  if (!tof.begin(Wire1)) {
    Serial.println("WARN: ToF init failed");
  } else {
    Serial.println("ToF init OK");
  }

  motor.begin();
  motor.enable();

  pid.configure({
    config::PID_KP,
    config::PID_KI,
    config::PID_KD,
    config::PID_I_LIMIT,
    config::PID_OUT_LIMIT
  });

  printHelp();
}

// -----------------------------
// Loop
// -----------------------------
void loop() {
  uint32_t now = millis();

  // -----------------------------
  // Time delta (shared)
  // -----------------------------
  static uint32_t lastTime = 0;
  float dt = (lastTime == 0) ? 0.0f :
             (now - lastTime) / 1000.0f;
  lastTime = now;

  // If the control loop hiccups badly, drop motor output immediately.
  if (dt > 0.2f) {
    manual_pwm_cmd = 0;
    pwm = 0;
    holding = false;
    pid.reset();
    motor.setPWM(0);
  }

  // -----------------------------
  // Read encoder (fast path)
  // -----------------------------
  int32_t counts = enc.readCounts();

  // -----------------------------
  // Read ToF (slow / optional)
  // -----------------------------
  lastToF = tof.read();

  bool tof_valid = lastToF.valid && lastToF.fresh;

  // -----------------------------
  // Estimator (runs EVERY loop)
  // -----------------------------
  lastEst = estimator.update(
    counts,
    tof_valid,          // fuse only if valid (tof_valid) or false
    lastToF.mm,
    dt
  );

  // -----------------------------
  // Serial
  // -----------------------------
  readSerialCommands();

  // -----------------------------
  // Safety
  // -----------------------------
  bool driverFaultActive = motor.faultActive();
  bool posBeyondRealLimit =
    (lastEst.pos_est_mm < config::STROKE_REAL_MIN_MM) ||
    (lastEst.pos_est_mm > config::STROKE_REAL_MAX_MM);

  if (posBeyondRealLimit) {
    posHardFaultLatched = true;
  }

  bool inhibitMotion = false;
  if (posHardFaultLatched) {
    manual_pwm_cmd = 0;
    pwm = 0;
    motor.disable();
    inhibitMotion = true;

    static uint32_t lastFaultPrint = 0;
    if (now - lastFaultPrint >= 1000) {
      lastFaultPrint = now;
      Serial.println("FAULT: Position exceeded 0..150mm, output disabled");
    }
  } else if (driverFaultActive) {
    manual_pwm_cmd = 0;
    pwm = 0;
    motor.setPWM(0);
    inhibitMotion = true;

    static uint32_t lastDriverWarnPrint = 0;
    if (now - lastDriverWarnPrint >= 1000) {
      lastDriverWarnPrint = now;
      Serial.println("WARN: Driver fault active, output inhibited");
    }
  }

  // -----------------------------
  // MANUAL CONTROL WITH SAFETY (runs EVERY loop)
  // -----------------------------
  if (!inhibitMotion && mode == ControlMode::MANUAL) {
    float pos = lastEst.pos_est_mm;
    bool extending = forward;

    float duty = static_cast<float>(manual_pwm_cmd) / static_cast<float>(PWM_MAX);
    duty *= computeSoftZoneScale(pos);

    if (isHardLimitHit(pos, extending)) {
      duty = 0.0f;
    }

    duty = constrain(duty, 0.0f, 1.0f);

    pwm = static_cast<int>(roundf(duty * static_cast<float>(PWM_MAX)));

    motor.setDirection(extending ?
      MotorDriver::Direction::EXTEND :
      MotorDriver::Direction::RETRACT);
    motor.setPWM(pwm);
  }

  // -----------------------------
  // AUTO CONTROL (runs EVERY loop)
  // -----------------------------
  else if (!inhibitMotion) {

    float pos = lastEst.pos_est_mm;
    float vel = lastEst.vel_est_mm_s;

    // --- target smoothing ---
    float max_step = 40.0f * dt;  // mm/s limit (tune this)

    float delta = target_mm_cmd - target_mm;

    if (fabs(delta) > max_step) {
      target_mm += (delta > 0 ? max_step : -max_step);
    } else {
      target_mm = target_mm_cmd;
    }

    target_mm = constrain(target_mm,
                          config::STROKE_HARD_MIN_MM,
                          config::STROKE_HARD_MAX_MM);

    float error = target_mm - pos;

    bool skipPid = false;

    // --- Holding logic ---
    if (holding) {
      if (fabs(error) > config::DRIFT_RESTART_MM) {
        holding = false;
      } else {
        pwm = 0;
        motor.setPWM(0);
        skipPid = true;
      }
    }

    // --- improved stop condition ---
    if (!skipPid) {
      bool near_target = fabs(error) <= config::POS_TOL_MM;
      bool slow = fabs(vel) < 0.5f;  // tighter threshold

      // check if we are NOT still moving toward the target
      bool not_pushing_toward_target =
        (error > 0 && vel <= 0) ||   // target ahead, but not moving toward it
        (error < 0 && vel >= 0);     // target behind, but not moving toward it

      if (near_target && slow && not_pushing_toward_target) {
        pwm = 0;
        motor.setPWM(0);
        holding = true;
        skipPid = true;
      }
    }

    if (!skipPid) {
      // --- PID ---
      float effort = pid.update(target_mm, pos, vel, dt);

      bool dir = (effort > 0);
      motor.setDirection(dir ?
        MotorDriver::Direction::EXTEND :
        MotorDriver::Direction::RETRACT);

      float duty = fabs(effort);

      if (duty > 0.0f) {
        duty = config::MIN_DUTY +
               (1.0f - config::MIN_DUTY) * duty;
      }

      duty *= computeSoftZoneScale(pos);

      if (isHardLimitHit(pos, dir)) {
        duty = 0.0f;
      }

      duty = constrain(duty, 0.0f, 1.0f);

      pwm = duty * PWM_MAX;
      motor.setPWM(pwm);
    }
  }

  // -----------------------------
  // STATUS (slow)
  // -----------------------------
  if (now - lastPrint >= 1000) {
    lastPrint = now;
    printStatus(lastEst);
  }
}
